211  void parse()
212  {
213  if (!m_is) throw Exception(Exception::KISAX_ERROR_BAD_STREAM);
214  if (!m_stop_flag) documentStart();
215  m_stop_flag = false;
216  while (m_is->get(m_current_char)&&!m_stop_flag)
217  {
218  m_state = m_states[m_state*MAXTOKEN+getSymbolType(m_current_char)];
219  if (m_state<0) throw Exception(Exception::KISAX_ERROR_PARSING);
220  (this->*(m_rules[m_state]))();
221  }
222  if (!m_stop_flag&&!m_is->eof()) throw Exception(Exception::KISAX_ERROR_BAD_STREAM);
223  documentEnd();
224  }

368  Symbol getSymbolType(char c)
369  {
370  if (std::isalpha(c)) return ALPHA;
371  if (std::isdigit(c)) return DIGIT;
372  switch (c)
373  {
374  case ' ' :
375  case '\n':
376  case '\t': return DELIMITER;
377  case '<' : return TAG_BEGIN;
378  case '>' : return TAG_END;
379  case '?' : return QUESTION;

381  case '\"': return QUOTE;
382  case '/' : return SLASH;
383  case '!' : return EXCLAMATION;
384  case '-' : return MINUS;
385  case '_' : return UNDERLINE;
386  case ':' : return COLON;
387  default : return ANOTHER;
